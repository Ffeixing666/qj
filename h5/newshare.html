// --- APK 上传与解析 (修改版：从 APK 文件内提取图标) --- document.getElementById('apkInput').onchange = async (e) => { const file = e.target.files[0]; if (!file) return; const tip = document.getElementById('status-tip'); tip.style.display = 'block'; tip.innerText = "正在深度解析 APK..."; try { const parser = new AppInfoParser(file); const info = await parser.parse(); // 1. 处理名称（原有逻辑保留） let finalName = info.name; if (info.application && info.application.label) { const labels = Array.isArray(info.application.label) ? info.application.label : [info.application.label]; finalName = labels.find(l => /[\u4e00-\u9fa5]/.test(l)) || labels[0]; } if (!finalName || String(finalName).startsWith('@')) finalName = info.package; // 2. 核心：从 APK 中提取图标文件流 const appIcon = document.getElementById('app-icon'); // 尝试从 parser 结果直接获取 base64 (AppInfoParser 通常自带这个功能) if (info.icon && info.icon.startsWith('data:image/')) { appIcon.src = info.icon; // 自动把提取到的图标同步上传到你的服务器，防止提交时 Base64 太大 uploadIcon(info.icon); } else { // 如果 parser 没提出来，我们用 JSZip 手动进 APK 找 const zip = await JSZip.loadAsync(file); // 常见的图标路径规律：res/drawable-xxxhdpi/icon.png 或 res/mipmap-xxxhdpi/ic_launcher.png // 这里我们找最像图标的文件 const iconEntry = Object.keys(zip.files).find(path => (path.includes('ic_launcher') || path.includes('icon')) && path.endsWith('.png') ); if (iconEntry) { const imgData = await zip.file(iconEntry).async("base64"); const iconBase64 = "data:image/png;base64," + imgData; appIcon.src = iconBase64; uploadIcon(iconBase64); } else { appIcon.src = 'data:image/svg+xml;utf8,'; } } // 3. 处理架构和 SDK (原有逻辑) let rawSdk = findMinSdk(info); rawSdk = typeof rawSdk === 'string' ? parseInt(rawSdk) : rawSdk; const androidVer = SDK_MAP[rawSdk] || "4.4"; const zip = await JSZip.loadAsync(file); const archs = new Set(); Object.keys(zip.files).forEach(p => { if (p.startsWith('lib/')) { const parts = p.split('/'); if (parts.length > 2) archs.add(parts[1]); } }); const archArray = Array.from(archs); const is64 = archArray.some(a => a.includes('64') || a === 'arm64-v8a'); const is32 = archArray.some(a => a.includes('v7a') || a === 'armeabi'); const bitDisplay = archArray.length === 0 ? "全架构兼容" : ((is64 && is32) ? "32/64位兼容" : (is64 ? "纯64位" : "纯32位")); // 4. UI 渲染 const sizeMB = (file.size / (1024 * 1024)).toFixed(2); document.getElementById('app-name').value = finalName; document.getElementById('app-pkg').innerText = info.package || 'unknown'; document.getElementById('v-text').innerText = info.versionName || '1.0.0'; document.getElementById('s-text').innerText = sizeMB + " MB"; document.getElementById('sdk-text').innerText = "Android " + androidVer + "+"; document.getElementById('arch-text').innerText = bitDisplay; // 5. APK 上传 (100MB 限制逻辑) let appLink = ''; if (parseFloat(sizeMB) <= 100) { appLink = await uploadApk(file); } else { document.getElementById('pan-section').style.display = 'block'; } document.getElementById('upload-zone').style.display = 'none'; tip.style.display = 'none'; document.getElementById('content-step').style.display = 'block'; apkInfo = { finalName, sizeMB, package: info.package, versionName: info.versionName, androidVer, bitDisplay, appLink }; } catch (err) { tip.innerHTML = "解析失败，请检查 APK 是否加密或损坏"; console.error(err); } }; // --- 图标自动预上传函数 --- async function uploadIcon(base64Data) { try { // 把 base64 转为 Blob 模拟文件上传 const arr = base64Data.split(','), mime = arr[0].match(/:(.*?);/)[1]; const bstr = atob(arr[1]), n = bstr.length, u8arr = new Uint8Array(n); while(n--) u8arr[n] = bstr.charCodeAt(n); const file = new File([u8arr], "icon.png", {type:mime}); const fd = new FormData(); fd.append('file', file); const res = await fetch('https://apirank.qujiaweb.top/?mode=upload_meituan', { method: 'POST', body: fd }); const data = await res.json(); uploadedIconUrl = data.full_url; console.log("图标预上传成功:", uploadedIconUrl); } catch (e) { console.warn("图标上传失败:", e); } }
